watermark operations
convert -size 250x50 xc:transparent -font Courier-bold -pointsize 20 -gravity  east -draw "fill white text 1,1 'LZ10376101090530'" -rotate 90 code.gif
convert -size 50x15 xc:transparent -font /Library/Fonts/Libian.ttc  -pointsize 20 -gravity center -draw "fill white text 1,1 '王'"  wang.gif
convert -size 50x15 xc:transparent -font /Library/Fonts/Libian.ttc  -pointsize 20 -gravity center -draw "fill white text 1,1 '宝'"  bao.gif
convert  -append code.gif wang.gif bao.gif  codewangbao.gif
composite  -quality 100 codewangbao.gif DEMO_before.jpg  DemoWatermark.jpg
